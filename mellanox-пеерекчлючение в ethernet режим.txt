находим карту меланов через lspci ее адрес 
замтем


# mstconfig -d 01:00.0 q | grep LINK_TYPE
         LINK_TYPE_P1                        VPI(3)          
         LINK_TYPE_P2                        VPI(3)

# sudo mstconfig -d 01:00.0 set LINK_TYPE_P1=2 LINK_TYPE_P2=2
