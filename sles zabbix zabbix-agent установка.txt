создаем конфиг для сервиса
надо именно руками потому что то что входит в комплект ооно для редхата а в слес 
оно неработает
	# cat /etc/init.d/zabbix-agent
#!/bin/sh
### BEGIN INIT INFO
# Provides:          zabbix-agent
# Required-Start:    $network $remote_fs
# Required-Stop:     $network $remote_fs
# Default-Start:     3 5
# Default-Stop:      0 1 2 6
# Short-Description: Zabbix agent
### END INIT INFO

. /lib/lsb/init-functions

CONF=/etc/zabbix/zabbix_agentd.conf
BIN=/usr/sbin/zabbix_agentd

case "$1" in
    start)
        echo -n "Starting Zabbix agent "
        start_daemon $BIN -c $CONF
        rc_status -v
        ;;
    stop)
        echo -n "Shutting down Zabbix agent "
        killproc -TERM $BIN
        rc_status -v
        ;;
    restart)
        $0 stop
        $0 start
        ;;
    status)
        echo -n "Checking for Zabbix agent "
        checkproc $BIN
        rc_status -v
        ;;
    *)
        echo "Usage: $0 {start|stop|status|restart}"
        exit 1
        ;;
esac
rc_exit



	# service zabbix-agent start
