и в его конфиг файле /etc/network/interfaces если там  в настроках порта
стоят линии закометированные например 


auto int0
iface int0 inet static
        address 192.168.2.50
	netmask 255.255.255.248
	gateway 192.168.2.49
        ovs_type OVSIntPort
        ovs_bridge vmbr0
        #ovs_extra set port int0 tag=100
	ovs_options tag=100
