(*)новых мудаков которых блочить добаавляем в файл /etc/ufw/before.rules (если юзаем ufw)

# cat  /etc/ufw/before.rules | grep -v '#' | head -n 20
*raw
:PREROUTING ACCEPT [0:0]
:OUTPUT ACCEPT [0:0]

-A PREROUTING -p udp --dport 53 -m string --hex-string "|00 00 ff 00 01|" --algo bm --from 40 -j DROP
-A PREROUTING -p tcp --dport 53 -m string --hex-string "|00 00 ff 00 01|" --algo bm --from 52 -j DROP


-A PREROUTING -s 20.17.63.63 -j DROP

COMMIT
